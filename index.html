<script>

    const monthsData = [

        { name: "一月", target: "存下 496 元", start: 1, end: 31 },

        { name: "二月", target: "存下 1274 元", start: 32, end: 59 },

        { name: "三月", target: "存下 2325 元", start: 60, end: 90 },

        { name: "四月", target: "存下 3165 元", start: 91, end: 120 },

        { name: "五月", target: "存下 4216 元", start: 121, end: 151 },

        { name: "六月", target: "存下 4995 元", start: 152, end: 181 },

        { name: "七月", target: "存下 6107 元", start: 182, end: 212 },

        { name: "八月", target: "存下 7068 元", start: 213, end: 243 },

        { name: "九月", target: "存下 7755 元", start: 244, end: 273 },

        { name: "十月", target: "存下 8959 元", start: 274, end: 304 },

        { name: "十一月", target: "存下 9585 元", start: 305, end: 334 },

        { name: "十二月", target: "存下 10850 元", start: 335, end: 365 }

    ];


    let checkedDays = JSON.parse(localStorage.getItem('365_savings_progress')) || [];

    const gridContainer = document.getElementById('months-grid');


    function renderGrid() {

        gridContainer.innerHTML = '';

        

        monthsData.forEach(month => {

            const col = document.createElement('div');

            col.className = 'month-column';


            const header = document.createElement('div');

            header.className = 'month-header';

            header.innerHTML = `<span>${month.name}</span><span class="month-target">${month.target}</span>`;

            col.appendChild(header);


            const daysList = document.createElement('div');

            daysList.className = 'days-list';


            for (let i = month.start; i <= month.end; i++) {

                const cell = document.createElement('div');

                cell.className = 'day-cell';

                cell.innerText = i;

                

                if (checkedDays.includes(i)) {

                    cell.classList.add('checked');

                }


                cell.addEventListener('click', () => toggleDay(i, cell));

                daysList.appendChild(cell);

            }


            col.appendChild(daysList);

            gridContainer.appendChild(col);

        });


        updateDashboard();

    }


    function toggleDay(dayAmount, element) {

        const index = checkedDays.indexOf(dayAmount);

        if (index === -1) {

            checkedDays.push(dayAmount);

            element.classList.add('checked');

        } else {

            checkedDays.splice(index, 1);

            element.classList.remove('checked');

        }

        

        localStorage.setItem('365_savings_progress', JSON.stringify(checkedDays));

        updateDashboard();

    }


    function updateDashboard() {

        const totalSaved = checkedDays.reduce((sum, val) => sum + val, 0);

        const daysCount = checkedDays.length;

        const progressPercent = ((daysCount / 365) * 100).toFixed(1);


        document.getElementById('total-saved').innerText = `$${totalSaved.toLocaleString()}`;

        document.getElementById('days-count').innerText = `${daysCount} / 365 天`;

        document.getElementById('progress-percent').innerText = `${progressPercent}%`;

    }


    renderGrid();

</script>
